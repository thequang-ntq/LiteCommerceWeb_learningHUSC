using Microsoft.AspNetCore.Mvc;

using MySolution.BusinessLayers;

using MySolution.DomainModels;

using System.Threading.Tasks;

namespace MySolution.HRM.Controllers

{

    public class EmployeeController : Controller

    {

        // Hiển thị danh sách

        // Trỏ về /Employee hoặc /Employee/Index

        public async Task<IActionResult> Index()

        {

            var model = await HRMService.ListEmployeesAsync();

            return View(model);

        }

        // Hiển thị chi tiết

        // Trả /Employee/Details/id

        public async Task<IActionResult> Details(string id)

        {

            var model = await HRMService.GetEmployeeAsync(id);

            return View(model);

        }

        // Hiển thị view thêm Employee

        [HttpGet]

        public IActionResult Create()

        {

            var model = new Employee();

            return View(model);

        }

        // Logic thêm

        [HttpPost]

        public async Task<IActionResult> Create(Employee model)

        {

            await HRMService.AddEmployeeAsync(model);

            // Về trang Index

            return RedirectToAction("Index");

        }

        // Hiển thị view sửa Employee

        [HttpGet]

        public async Task<IActionResult> Update(string id)

        {

            var model = await HRMService.GetEmployeeAsync(id);

            return View(model);

        }

        // Logic sửa

        [HttpPost]

        public async Task<IActionResult> Update(Employee model)

        {

            await HRMService.UpdateEmployeeAsync(model);

            // Về trang Index

            return RedirectToAction("Index");

        }

        // Xóa dựa vào EmployeeId, không thêm giao diện mới, bấm vào là xóa luôn

        public async Task<IActionResult> Delete(string id)

        {

            await HRMService.DeleteEmployeeAsync(id);

            // Về trang Index

            return RedirectToAction("Index");

        } 

    }

}
